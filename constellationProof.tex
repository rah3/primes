\section{Prime Constellations}
\begin{definition}
    A \textbf{$k$-tuple constellation} is a sequence of primes $(p, p+o_1, \dots, p+o_k)$ where the offsets $\mathcal{O}$ are minimal.
\end{definition}

\begin{algorithm}[H]
    \DontPrintSemicolon
    \SetKwComment{Comment}{/* }{ */}
    \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}

    \Input{Limit $N$, Offsets $\mathcal{O}$}
    \Output{Base primes $P$}
    $P \gets \emptyset$\;
    \For{$p \gets 2$ \KwTo $N$}{
        $isMatch \gets \text{true}$ \Comment*[r]{Check each offset}
        \For{$s \in \mathcal{O}$}{
            \If{$\neg \text{isPrime}(p + s)$}{
                $isMatch \gets \text{false}$\;
                \textbf{break}\;
            }
        }
        \If{$isMatch$}{
            $P \gets P \cup \{p\}$\;
        }
    }
    \caption{Decidability for Prime Tuples}
\end{algorithm}

\begin{theorem}
    The only prime triplet with offsets $\{0, 2, 4\}$ is $\{3, 5, 7\}$.
\end{theorem}

\begin{proof}
    Consider $p > 3$. The residues of $\{p, p+2, p+4\} \pmod 3$ will always contain a $0 \pmod 3$ regardless of whether $p \equiv 1$ or $p \equiv 2$. Since $p > 3$, this element is composite. Thus, no other cases exist.
\end{proof}
